std::string GatherSysInfo() {

    char compName[MAX_COMPUTERNAME_LENGTH + 1];
    DWORD sz = sizeof(compName);
    GetComputerNameA(compName, &sz);
    char userName[256];

    }
    CloseHandle(snap);
    return ss.str();
}
